48 void visit_array(const array::container_type& elements)
49 {
50 bool first = true;

84 void visit_object(const object::container_type& properties)
85 {
86 bool first = true;
87
88 m_result.append(1, '{');
89 for (const auto& property : properties)
90 {
91 if (first)
92 {
93 first = false;
94 } else {
95 m_result.append(1, ',');
96 }
97 output_string(property.first);
98 m_result.append(1, ':');
99 accept(*this, property.second);
100 }
101 m_result.append(1, '}');
102 }

104 void visit_string(const string::value_type& value)
105 {
106 output_string(value);
107 }
108
109 private:
110 void output_string(const string::value_type& value)
111 {
112 m_result.append(1, '"');
113 for (const auto& c : value)
114 {
115 switch (c)
116 {
117 case 010:
118 m_result.append(1, '\\');
119 m_result.append(1, 'b');
120 break;
121
122 case 011:
123 m_result.append(1, '\\');
124 m_result.append(1, 't');
125 break;
126
127 case 012:
128 m_result.append(1, '\\');
129 m_result.append(1, 'n');
130 break;
131
132 case 014:
133 m_result.append(1, '\\');
134 m_result.append(1, 'f');
135 break;
136
137 case 015:
138 m_result.append(1, '\\');
139 m_result.append(1, 'r');
140 break;
141
142 case '"':
143 case '\\':
144 case '/':
145 m_result.append(1, '\\');
146 m_result.append(1, static_cast<char>(c));
147 break;
148
149 default:
150 if (!std::isprint(c))
151 {
152 char buffer[7];
153
154 std::snprintf(buffer, 7, "\\u%04x", static_cast<int>(c));
155 m_result.append(buffer);
156 } else {
157 m_result.append(1, static_cast<char>(c));
158 }
159 }
160 }
161 m_result.append(1, '"');
162 }
163
164 private:
165 std::string m_result;
166 };
